﻿מעין שלוש

על מזונות, יין ותירוש ועל פירות חמשת הממינים

בָּרוּךְ אַתָּה יְהֹוָה אֱלֹהֵֽינוּ תבהוהכ מֶֽלֶךְ הָעוֹלָם,

עַל־הַמִּחְיָה וְעַל־הַכַּלְכָּלָהמזונות;

עַל־הַגֶּֽפֶן וְעַל־פְּרִי הַגֶּֽפֶןיין;

עַל־הָעֵץ וְעַל־פְּרִי הָעֵץמשבעת המינים;

וְעַל־תְּנוּבַת הַשָּׂדֶה, וְעַל־אֶֽרֶץ חֶמְדָּה, טוֹבָה וּרְחָבָה, שֶׁרָצִֽיתָ וְהִנְחַֽלְתָּ לַאֲבוֹתֵֽינוּ , לֶאֱכֹל מִפִּרְיָהּ , וְלִשְׂבַֹּֽ מִטּוּבָהּ. רַחֵם יְהֹוָה אֱלֹהֵֽינוּ עָלֵֽינוּ , וְעַל־יִשְׂרָאֵל עַמָּךְ, וְעַל־יְרוּשָׁלַֽיִם עִירָךְ, וְעַל־הַר צִיּוֹן מִשְׁכַּן כְּבוֹדָךְ, וְעַל־מִזְבָּחָךְ, וְעַל־הֵיכָלָךְ. וּבְנֵה יְרוּשָׁלַֽיִם עִיר הַקֹּֽדֶשׁ, בִּמְהֵרָה בְּיָמֵֽינוּ. וְהַעֲלֵֽנוּ לְתוֹכָהּ , וְשַׂמְּחֵֽנוּ בְּבִנְיָנָהּ , וּנְבָֽרְכָֽךְ עָלֶֽיהָ בִּקְדֻשָּׁה וּבְטָהֳרָה.

וְזָכְרֵֽנוּ לְטוֹבָה בְּיוֹם רֹאשׁ חֹֽדֶשׁ הַזֶּה. בראש־חודש;

וְשַׂמְּחֵֽנוּ בְּיוֹם (חַג הַמַּצּוֹת \ חַג הַסֻּכּוֹת) > הַזֶּה, בְּיוֹם מִקְרָא קֹֽדֶשׁ הַזֶּה. בחול־המועד;

כִּי אַתָּה טוֹב וּמֵטִיב לַכֹּל, וְנֽוֹדֶה לְּךָ יְהֹוָה אֱלֹהֵֽינוּ , עַל־הָאָֽרֶץ,

.[כַּלְכָּלָתָהּ] וְעַל־הַכַּלְכָּלָה [מִחְיָתָהּ] וְעַל־הַמִּחְיָה מזונות;

.[גַפְנָהּ] וְעַל־פְּרִי הַגֶּֽפֶן יין;

.[פֵּרוֹתֶֽיהָ] וְעַל־הַפֵּרוֹת שבעת המינים;

בָּרוּךְ אַתָּה יְהֹוָה, עַל הָאָֽרֶץ

.[כַּלְכָּלָתָהּ] וְעַל־הַכַּלְכָּלָה [מִחְיָתָהּ] וְעַל־הַמִּחְיָה מזונות;

.[גַפְנָהּ]וְעַל־פְּרִי הַגֶּֽפֶן יין;

.[פֵּרוֹתֶֽיהָ] וְעַל־הַפֵּרוֹת שבעת המינים;

הָרַחֲמָן הוּא יְחַיֵּֽנוּ וִיזַכֵּֽנוּ וִיקָֽרְבֵֽנוּ לִימוֹת הַמָּשִֽׁיחַ וּלְבִנְיַן בֵּית הַמִּקְדָּשׁ, וּלְחַיֵּי הָעוֹלָם הַבָּא. ת מַלְכּ֑וֹ, וְעֹֽשֶׂה־מִגְדּ֖וֹל יְשׁוּע֣וֹ ;ובחול המועדבראש חודש חֶ֧סֶד לִמְשִׁיח֛וֹ - לְדָוִ֥ד וּלְזַרְע֖וֹ עַד־עוֹלָֽם:

מַגְדִּיל֮ יְשׁוּע֣וֹת מַלְכּ֑וֹ, וְעֹֽשֶׂה־חֶ֧סֶד לִמְשִׁיח֛וֹ - ל ;בימים אחריםלְדָוִ֥ד וּלְזַרְע֖וֹ עַד־עוֹלָֽם:

כְּ֭פִירִֽים לֹֽא־ רָשׁ֣וּ וְרָעֵ֑בוּ , וְדֹֽרְשֵׁ֥י יְ֝הֹוָ֗היַחְסְר֥וּ כָל־טֽוֹב: נַ֤עַר ׀ הָיִ֗יתִי גַּם־זָ֫קַ֥נְתִּי, וְֽלֹא־רָ֭אִֽיתִי צַדִּ֣יק נֶֽעֱזָ֑ב וְ֝זַרְע֗וֹ מְבַקֶּשׁ־לָֽחֶם: כָּל־הַ֭יּֽוֹם חוֹנֵ֣ן וּמַלְוֶ֑ה, וְ֝זַרְע֗וֹ לִבְרָכָֽה:

מַה־שֶּׁאָכַֽלְנוּיִהְיֶה לְשָׂבְעָה, וּמַה־שֶּׁשָּׁתִֽינוּ יִהְיֶה לִרְפוּאָה, וּמַה־שֶּׁהוֹתַֽרְנוּ יִהְיֶה לִבְרָכָה, כְּדִכְתִיב; וַיִּתֵּ֧ן לִפְנֵיהֶ֛ם, וַיֹּֽאכְל֥וּ וַיּוֹתִ֖רוּ כִּדְבַ֥ר יְהֹוָֽה: בְּרוּכִ֣ים אַ֭תֶּֽם לַֽיהֹוָ֑ה, עֹ֝שֵׂ֗ה שָׁמַ֥יִם וָאָֽרֶץ: בָּר֣וּךְ הַגֶּ֔בֶר אֲשֶׁ֥ר יִבְטַ֖ח בַּֽיהֹוָ֑ה, וְהָיָ֥ה יְהֹוָ֖ה מִבְטַחֽוֹֹ: יְֽהֹוָ֗ה עֹ֭ז לְעַמּ֣וֹ יִתֵּ֑ן, יְהֹוָ֓ה ׀ יְבָרֵ֖ךְ אֶת־עַמּ֣וֹ בַשָּׁלֽוֹם:

עֹשֶׂה שָׁלוֹם בִּמְרוֹמָיו, הוּא בְּרַחֲמָיו יַעֲשֶׂה שָׁלוֹם עָלֵֽינוּ וְעַל־כָּל־עַמּוֹ יִשְׂרָאֵל, וְאִמְרוּ אָמֵן.
